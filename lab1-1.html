def filterSignal(x, Fs, low, high, order):
#   Each column in x is one
#   recording channel. Fs is the sampling frequency. low and high specify the passband in Hz.
    relativelow=low/(Fs/2)
    relativehigh=high/(Fs/2)
    Wn=[relativelow,relativehigh]
    # filter the signal independently for each column in x as each is one channel
    
    
    sos = signal.butter(order, Wn, 'bp', fs=Fs, output='sos')
    print(sos.shape)
    y = signal.sosfilt(sos, x, axis=0)
    return(y)
